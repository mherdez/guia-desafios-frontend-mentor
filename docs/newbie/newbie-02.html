# Diseño responsivo (2da. parte del desafío)

# @media()

@media (max-width: 768px)  {
  body {
    padding: 2rem;
  }
  .card {
    flex-direction: column;
    text-align: center;

    &-body {
      &-text {
        font-size: 1.2rem;
        font-kerning: normal;
        line-height: 2rem;
      }
      
      &-section {
        flex-direction: column;

        &-stats {
          margin-bottom: 2.2rem;
        }
      }
      
    }

  }
}
